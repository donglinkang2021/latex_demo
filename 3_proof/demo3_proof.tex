\documentclass{article}
\usepackage{amsmath}
\usepackage{ctex}
\newtheorem{theorem}{Theorem} % 定义定理环境，定理与引理的编号相同
\newtheorem{lemma}{Lemma} % 定义引理环境，引理与定理的编号相同
\newtheorem{proof}{Proof}[section] % 使证明环境的编号与章节编号相关

\begin{document}

    \section{theorem}

    \begin{theorem}
        This is a theorem. (定理)
    \end{theorem}

    \begin{lemma}
        This is a lemma. (引理)
    \end{lemma}

    \section{Proof}

    \begin{proof}
        This is proof. (证明)
    \end{proof}


\end{document}
